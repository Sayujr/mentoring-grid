Set Up Google Sheets Live Push

Follow these steps once — data will push automatically whenever a quote is saved.

  1. Open Google Sheets and create a new spreadsheet.
  2. Go to Extensions → Apps Script.
  3. Paste the script below, click Save, then Deploy → New deployment → Web app.
  4. Set Execute as: Me and Who has access: Anyone. Click Deploy and copy the URL.
  5. Paste that URL into the Google Sheets URL field above.
function doPost(e) {
  const s = SpreadsheetApp.getActiveSheet();
  const d = JSON.parse(e.postData.contents);
  s.appendRow([
    new Date(), d.initials, d.quadrant,
    d.x + '%', d.y + '%', d.quote
  ]);
  return ContentService
    .createTextOutput('OK')
    .setMimeType(ContentService.MimeType.TEXT);
}
